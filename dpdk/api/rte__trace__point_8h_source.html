298 static __rte_always_inline void *
299 __rte_trace_mem_get(uint64_t in)
300 {
301  struct __rte_trace_header *trace =
302  (struct __rte_trace_header *)(RTE_PER_LCORE(trace_mem));
303  const uint16_t sz = in & __RTE_TRACE_FIELD_SIZE_MASK;
304 
305  /* Trace memory is not initialized for this thread */
306  if (unlikely(trace == NULL)) {
307  __rte_trace_mem_per_thread_alloc();
308  trace = (struct __rte_trace_header *)(RTE_PER_LCORE(trace_mem));
309  if (unlikely(trace == NULL))
310  return NULL;
311  }
312  /* Check the wrap around case */
313  uint32_t offset = trace->offset;
314  if (unlikely((offset + sz) >= trace->len)) {
315  /* Disable the trace event if it in DISCARD mode */
316  if (unlikely(in & __RTE_TRACE_FIELD_ENABLE_DISCARD))
317  return NULL;
318 
319  offset = 0;
320  }
321  /* Align to event header size */
322  offset = RTE_ALIGN_CEIL(offset, __RTE_TRACE_EVENT_HEADER_SZ);
323  void *mem = RTE_PTR_ADD(&trace->mem[0], offset);
324  offset += sz;
325  trace->offset = offset;
326 
327  return mem;
328 }
