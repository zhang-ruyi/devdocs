631 static inline struct rte_eth_dev *
632 ETHDEV(struct pmd_internals *softnic)
633 {

635  int status;
636 
637  if (softnic == NULL)
638  return NULL;
639 
640  status = rte_eth_dev_get_port_by_name(softnic->params.name, &port_id);
641  if (status)
642  return NULL;
643 
644  return &rte_eth_devices[port_id];
645 }
