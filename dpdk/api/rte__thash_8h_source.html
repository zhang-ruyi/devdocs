181  const uint8_t *rss_key)
182 {

184 
185  for (j = 0; j < input_len; j++) {

187  i = rte_bsf32(map);
188  ret ^= rte_cpu_to_be_32(((const uint32_t *)rss_key)[j]) << (31 - i) |
189  (uint32_t)((uint64_t)(rte_cpu_to_be_32(((const uint32_t *)rss_key)[j + 1])) >>
190  (i + 1));
191  }
192  }
